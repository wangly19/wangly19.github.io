# 类型

let nan = NaN // number 类型
let b = 'asd' // string 类型
let c = true // boolean类型

# 数据类型判断

常用的数据数据类型判断类型通常使用typeof，instanceof来判断类型。但是两者之间是有区别的，typeof只能判断一些基本类型，在判断引用类型的时候，都是输出Object，所以碰到引用类型就需要请它的师兄instanceof出场了。甚至还有最后大杀器Object.prototype.toString()，

typeof 'aabb'      // 'string'
typeof true        // 'boolean'
typeof 10          // 'number'
typeof Symbol()    // 'symbol'
typeof null        // 'object' 无法判定是否为 null
typeof undefined   // 'undefined'
typeof []          // 'uobject
typeof {}          // 'object
typeof function (){}// 'function

(()=>{}) instanceof Function   // true

# 闭包

闭包是函数和声明函数的语法环境组合，在全局空间中开辟一份内部空间

//  简单闭包
(function () {
  var a  = 2
  function add() {
    var b = 2
    var sum = a + b
    console.log(sum)
  }
  add() // 4
})()

闭包最大的作用就是隐藏变量，内部函数总是可以访问其所在的外部函数中声明的参数和变量，因此私有变量、特权变量、储存变量就是基于这个实现的

// bean private value
function Data() {
  var name = 'Test'
  this.getName () {
    return name
  }
  this.setName (value){
    name = value    
  }
}
// new实例化
const data = new Data()
// 输出name
console.log(data.getName()) //=> Test
// 设置name在输出
data.setName('wang')
console.log(data.getName()) // =>wang
console.log(name)  //  => name没有被声明

# this的 了解

this的只想只有在程序运行后确定的，同时this在不同的指向也遵从规则，默认情况下是指向全局对象，比如浏览器中的window

# call/bind/apply

call、apply、bind的作用主要是用来改变this的指向。

# call

以参数列表的方式进行传入，然后改变this指向，一个案例

let myObject = {
  name: 'wang' // name => wang
}

const myFoo = function () {
  /* 创建一个函数 */
  console.log(this.name)  //输出this.name
}

/* 改变this指向 */
myFoo.call(myObject)  // 第一个参数为对象
// 这个DEMO输出的是wang，因此可以看到this指向是myObject所以this.name会是'wang'

那么call是如何去实现的呢？

Function.prototype.myCall = function (context,...params) { // => 传入this和参数
  // 当传入context不是对象，就直接给它为null
  if (typeof context === 'object'){
    context = context || window
  }else{
    context = null
  }

  // 使用smybol变量，保持方法唯一
  let MYFUN = Symbol()
  context[MYFUN] = this // 调用方法的this
  context[MYFUN](...params) // 参数添加
  console.log(context[MYFUN]) // 删除临时对象
  delete context[MYFUN]
}

# bind

bind非常有意思，调用它并且传入需要绑定的对象，完成会返回一个改变this的函数（Function）

let myObject = {
  name: 'wang' // name => wang
}

const myFoo = function () {
  /* 创建一个函数 */
  console.log(this.name)  //输出this.name
}

let myFooRtn = myFoo.bind() // => wang
myFooRtn() // => 新的函数对象在传入对应参数

# apply

apply和call是一样改变this且执行函数，只是额外参数是以数组形式传入的

let myObject = {
  name: 'wang' // name => wang
}

const myFoo = function (width,height) {
  /* 创建一个函数 */
  console.log(this.name)  //输出this.name
  console.log(width)  //输出this.name
  console.log(height)  //输出this.name
}

/* 改变this指向 */
myFoo.apply(myObject,[12,'19cm'])  // => wang 12 19cm
