        # 冒泡排序

        bubbleSort(array){
        // 判断是否为数组
        if(typeof array === 'object' && array instanceof Array){
            // 复制数组，不直接操作原来的参数
            let newArr = [].concat(array)
            // 判断数组是否只有一个单位，如果是的话返回原来的数组
            if(newArr.length === 1) { return newArr }
            for(let i = 0 ; i < newArr.length - 1 ; i++){
                for(let j = 0 ; j < newArr.length - 1 - i ; j++){
                    // 如果当前数比下一位数大的话，将两个人的位置进行调换
                    if(newArr[j] > newArr[j + 1]){
                        const value = newArr[j]
                        newArr[j] = newArr[j + 1]
                        newArr[j + 1] = value
                    }
                }
            }
            // 返回数组
            return newArr
        }
        return '参数不是一个数组'
    }

        uniqArray(array){
        // 判断是否为数组
        if(typeof array === 'object' && array instanceof Array){
            let arr = []
            for(let index = 0 ; index < array.length ; index++){
                // 如果新的数组不包含该项，则push
                if(!arr.includes(array[index])){
                    arr.push(array[index])
                }
            }
            return arr
        }
    }

        # 深克隆

        shallowClone(params){
        if(typeof params !== 'object') return params
        if(params === null) return null
        // 定义变量
        let result = null
        // 判定参数类型
        if(params instanceof Array){
            // 如果参数是数组
            result = new Array()
            for(let item in params) result.push(params[item])
        }else 
        
        // 判定参数类型
        if(params instanceof Object){
            //如果参数是对象
            result = {}
            for(let item in params){
                result[item] = params[item]
            }
            
        }
        return result
    }

         getUrlParams(URL){
        // 将参数分割成字符串  name = 'wangly'数组
        let paramsArray = URL.split('?')[1].split('&')
        //创建存储空对象
        let paramsObject = {}
        // 循环
        for (let item of paramsArray){
            // 分割 = 号 
            let arr = item.split('=')
            paramsObject[arr[0]] = arr[1]
        }
        return paramsObject || null
    }

        # 设置参数

        setUrlParams(URL,data){
        // 添加 ?
        let url = URL + '?'
        // 判断类型
        if(data instanceof Object){
            for (let key in data){
                // 添加字符
                url = url+= key + '=' + data[key] + "&"
            }
            // 去除最后一个多余的&
            url = url.substring(0,url.length -1)
            // 返回结果
            return url || URL
        }
    }

        debounce(foo , whit){
        // 存储timeout 变量
        let timeout = null
        return function (){
            if(timeout !== null){
                // timeout被实例化了就清空它，状态一直为新的
                clearTimeout(timeout)
            }
            // 设置
            timeout = setTimeout(foo,whit)
        }
