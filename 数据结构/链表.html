# 单项链表

链表是编程里面非常重要的一个数据结构，它如同一个火车一般一节挂载着一节，从透到尾。

// 单向链表
class LinkedList{
    
    constructor(){
        this.length = 0
        this.head = null
        this.Node = function(element){
            this.element = element
            this.next = null
        }
    }

    // 插入节点
    append(value){
        let node = new this.Node(value)
        let current = null
        if(this.head === null){
            this.head = node
        }else{
            current = this.head
            while(current.next){
                current = current.next
            }
            current.next = node
        }
        this.length++
    }

    // 插入节点
    insert(position,value){
         //检查越界
         if(position > -1 && position < this.length){
             // new Node
             let node = new this.Node(value)
             let current = this.head
             let backup = null
             let index = 0
             if(position === 0){
                 node.next = current
                 this.head = node
             }else{
                 for(let i = 0 ; i < position ; i++){
                     // 循环
                     backup =   current
                     current = current.next
                 }
                 node.next = current
                 backup.next = node
             }
             return true
         }else {
             return false
         }
    }

    // 删除节点
    remove(position){
        // 判断节点
        if(position > -1 && position < this.length){
            let current = this.head
            let backup = null
            if(position === 0){
                this.head = current.next
            }else {
                for(let i = 0 ; i < position ; i++){
                    // 备份头部
                    backup = current
                    current = current.next
                }
                backup.next = current.next
            }
            this.length--
            return current.element
        }else {
            return false
        }
    }

    // 查询链表是否存下该值
    indexOf( value ){
        let current = this.head
        for(let i = 0 ; i < this.length ; i++){
            if(current.element === value){
                let LIST_INDEX = i
                return LIST_INDEX
            }
            current = current.next
        }
        return -1
    }

    // 判断链表是否为空
    isEmpty(){
        return this.length === 0 || false
    }

    // 链表长度
    size(){
        return this.length
    }
    
    // 清空链表
    clearList(object){
    }


    // 输出链表内容
    output(){
        let current = this.head
        let LIST_STR = ''
        let OR = '==>'
        while(current){
            if(current.next !== null){
                LIST_STR = LIST_STR + current.element + OR
            }else {
                LIST_STR = LIST_STR + current.element
            }
            current = current.next
        }
        return LIST_STR || null
    }
}
